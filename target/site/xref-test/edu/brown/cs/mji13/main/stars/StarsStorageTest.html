23    /**
24     * Sets up the Generator
25     */
26    @Before
27    public void setUp() {
28      storage = new StarStorage("Test");
29      ArrayList<Star> starsList = new ArrayList<Star>();
30      Star[] starArray = {new Star("1", "1", Arrays.asList(0.0, 0.0, 0.0)),
31          new Star("2", "2", Arrays.asList(10.0, 6.0, 5.0)),
32          new Star("3", "3", Arrays.asList(3.1, 1.5, -6.7)),
33          new Star("4", "4", Arrays.asList(-5.0, -6.0, -7.0)),
34          new Star("5", "5", Arrays.asList(-10.56, -13.0, 12.0)),
35          new Star("6", "6", Arrays.asList(0.5, 0.0, 0.0)),
36          new Star("7", "7", Arrays.asList(-50.0, -50.0, -50.0)),
37          new Star("8", "8", Arrays.asList(5.0, 6.0, 7.0)),
38          new Star("9", "B", Arrays.asList(3.0, 4.0, 5.0)),
39          new Star("10", "A", Arrays.asList(10.0, 20.0, 3.0))};
40      starsList.addAll(Arrays.asList(starArray));
41      storage.setListToStarsMap(starsList);
42    }
43  
44    /**
45     * Resets the Configuration.
46     */
47    @After

52    /**
53     ** Testing Map Functionalities
54     */
55    @Test
56    public void testStarsMap() {
57      setUp();
58      // Testing if name is in storage
59      assertTrue(storage.isNameInStorage("2"));
60      assertTrue(storage.isNameInStorage("B"));
61      assertTrue(storage.isNameInStorage("A"));
62      assertFalse(storage.isNameInStorage("67475"));
63      // Empty String
64      assertFalse(storage.isNameInStorage(""));
65  
66      // Getting Stars from the Map
67      assertEquals(storage.getStarFromMap("A").getStarID(), "10");
68      assertEquals(storage.getStarFromMap("B").getStarID(), "9");
69      assertNull(storage.getStarFromMap("C"));
70  
71      // Finding the Set of Names
72      Set<String> nameSet = storage.getStarNames();
73      assertTrue(nameSet.contains("2"));
74      assertTrue(nameSet.contains("4"));
75      assertTrue(nameSet.contains("A"));
76  
77      tearDown();
78    }

84    public void testStarsHTML() {
85      setUp();

87      Star[] starArray = {new Star("1", "1", Arrays.asList(0.0, 0.0, 0.0)),
88          new Star("2", "2", Arrays.asList(10.0, 6.0, 5.0)),
89          new Star("3", "3", Arrays.asList(3.1, 1.5, -6.7))};
90      starsList.addAll(Arrays.asList(starArray));
91      String table = storage.starListToHTML(starsList);
92  
93      assertEquals(table, "<table><tr><th>StarID</th><th>ProperName</th>" +
94          "<th>Coordinates (X, Y, Z)</th>" +
95          "</tr><tr><td>1</td><td>1</td><td>[0.0, 0.0, 0.0]</td></tr>" +
96          "<tr><td>2</td><td>2</td><td>[10.0, 6.0, 5.0]</td></tr>" +
97          "<tr><td>3</td><td>3</td><td>[3.1, 1.5, -6.7]</td></tr></table>");
98  
99      // Empty Table
100     String emptyTable = storage.starListToHTML(new ArrayList<>());

103             "</tr></table>");
