16    @Test
17    public void testNodeGetSet() {
18      KDNode<Star> sampleNode = new KDNode<Star>(
19          new Star("Abc", "My life is a joke", Arrays.asList(1.6, 5.4, 3.2)));
20  
21      sampleNode.setLeft(new KDNode<Star>(
22          new Star("1", "3", Arrays.asList(1.6, 0.4, 3.7))));
23      sampleNode.setRight(new KDNode<Star>(
24          new Star("3", "35", Arrays.asList(-8.9, 4.4, 3.6))));
25  
26      assertEquals(sampleNode.getItem().getStarID(), "Abc");
27      assertEquals(sampleNode.getLeft().getItem().getStarID(), "1");
28      assertEquals(sampleNode.getRight().getItem().getStarID(), "3");
29  
30      sampleNode.setItem(new Star("6", "1", Arrays.asList(1.6, 5.4, 3.2)));
31      assertEquals(sampleNode.getItem().getStarID(), "6");
32    }
