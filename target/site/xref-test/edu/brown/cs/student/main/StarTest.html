17     */
18    @Test
19    public void testStarDistanceTo() {
20      Star star1 = new Star("Abc", "My life is a joke", "2", "3", "4");
21      // Distance Normal
22      assertEquals(star1.distanceTo(0, 13, 4), 10.198, 0.01);
23      // Distance to itself
24      assertEquals(star1.distanceTo(2, 3, 4), 0, 0.01);
25      // Distance to Negatives
26      assertEquals(star1.distanceTo(-1, 2.0, -3), 7.6811, 0.01);
27      // Distance to Zero
28      assertEquals(star1.distanceTo(0, 0, 0), 5.38516, 0.01);
29    }
