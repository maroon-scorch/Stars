136   /**
137    * Determines if the String Date is in the format mm/dd/yy, zero can be omitted.
138    *
139    * @param input - text to be passed in
140    * @return True if format is matched and date is in range
141    */
142   public Boolean isMockDate(String input) {
143     boolean dateIsInRightFomat = input.matches(
144         "([1-9]|0[1-9]|1[012])[/]([1-9]|0[1-9]|[12][0-9]|3[01])[/]\\d\\d\\d\\d");
145     return (dateIsInRightFomat && isMockDateInRage(input));
146   }

148   /**
149    * Given a string date in the format "mm/dd/yy" checks if the day and months are in range.
150    *
151    * @param input - text to be passed in
152    * @return True if date is in range
153    */
154   private Boolean isMockDateInRage(String input) {
155     String[] dateArray = input.split("/");
156     int month = Integer.parseInt(dateArray[0]);
157     int day = Integer.parseInt(dateArray[1]);
158     int year = Integer.parseInt(dateArray[2]);
159 
160     int monthDays = monthToDays.get(month);
161     return ((day <= monthDays)
162         || ((month == 2) && (day == (monthToDays.get(2) + 1)) && (year % 4 == 0)));
163   }

165   /**
166    * Given a string email, checks if the email is valid.
167    *
168    * @param input - text to be passed in
169    * @return True if email is in the regex form specified = [username]@[domain]
170    */
171   public Boolean isMockEmail(String input) {
172     // Referenced from https://www.tutorialspoint.com/validate-email-address-in-java
173     return (input.isEmpty()
174         || input.matches("^[\\w-_\\.+]*[\\w-_\\.]\\@([\\w-_]+\\.)+[\\w]+[\\w]$"));
175   }

183   public Boolean isMockStreetAddress(String input) {
184     return (input.isEmpty()
185         || input.matches("(\\d+)[ ]([A-Za-z0-9\\s]+)"));
