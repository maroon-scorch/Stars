90    /**
91     * Executes the naive_radius Command.
92     * If successful, prints out every stars with radius less than or equal to that of
93     * the specified radius.
94     *
95     * Note: TA Colton said that the randomization is meant for tiebreakers to include on the list
96     * since naive_radius gets all the stars within range in, there are no tiebreakers to decide
97     * which "star" gets in.
98     * @param args - the list of arguments to be operated on.
99     * @return the Arraylist of starIDs to be print out if Successful. If unSuccessful,
100    * prints the error messages instead.
101    */
102   public ArrayList<String> execute(ArrayList<String> args) {
103     ArrayList<String> messages = new ArrayList<>();
104     hasErrorOccurred = false;
105     answers.clear();
106 
107     // If the current file is empty, print an error
108     if (starStorage.getFileName().length() == 0) {
109       hasErrorOccurred = true;
110       messages.add("ERROR: No file has been loaded yet");
111       return messages;
112     }
113 
114     // If no methods are matched, exits the command
115     Optional<String> opMethodName = matchArgsToMethod(args);
116     if (opMethodName.isEmpty()) {
117       hasErrorOccurred = true;
118       messages.add(argsValidator.getErrorMessage());
119       return messages;
120     }
121 
122     // If a method name has been found, determines which method it maps to and executes it
123     String methodName = opMethodName.get();
124     ArrayList<Star> slist = starStorage.getStarsList();
125     switch (methodName) {
126       case "naive_radius_2":
127         double radius = Double.parseDouble(args.get(0));
128         String sName = args.get(1);
129         String sStarNoQuotes = sName.substring(1, sName.length() - 1);
130         if (starStorage.isNameInStorage(sStarNoQuotes)) {
131           answers = performNaiveRadius(radius, sStarNoQuotes, slist);
132           answers.forEach((str) -> messages.add(str.getStarID()));
133         } else {
134           hasErrorOccurred = true;
135           messages.add(String.format("ERROR: No Stars with name \"%s\" "
136               + "is found or name is empty", sStarNoQuotes));
137         }
138         return messages;
139 
140       case "naive_radius_4":
141         double dRadius = Double.parseDouble(args.get(0));
142         double dPosX = Double.parseDouble(args.get(1));
143         double dPosY = Double.parseDouble(args.get(2));
144         double dPosZ = Double.parseDouble(args.get(3));
145         answers = performNaiveRadius(dRadius, dPosX, dPosY, dPosZ, slist);
146         answers.forEach((str) -> messages.add(str.getStarID()));
147         return messages;
148 
149       default:
150         // Only occurs if Programmer made an Error
151         System.out.println("ERROR: Hashmap reqInfoMaps has unregistered names, "
152             + "shouldn't have reached here");
153         return new ArrayList<>();
154     }
155   }

179   /**
180    * Finds all stars whose distance to the star specified with the name is less than or equal to
181    * the radius given.
182    *
183    * @param r the radius given
184    * @param name the name of the star
185    * @param alos the list of stars to search through
186    * @return the list of stars within range inclusive.
187    */
188   public ArrayList<Star> performNaiveRadius(double r, String name, ArrayList<Star> alos) {
189     // Gets the Star from StarStorage, because the program checked if the name is in the
190     // storage before, error handling is taken care of already.
191     Star presentStar = starStorage.getStarFromMap(name);
192 
193     double selectedX = presentStar.getCoordinate(0);
194     double selectedY = presentStar.getCoordinate(1);
195     double selectedZ = presentStar.getCoordinate(2);
196 
197     ArrayList<Star> template = new ArrayList<>(alos);
198     // Removes the star from the list returned after search.
199     template.removeIf(star -> star.getName().equals(name));
200 
201     return performNaiveRadius(r, selectedX, selectedY, selectedZ, template);
202   }

204   /**
205    * Finds all stars whose distance to the coordinate (x, y, z) is less than or equal to
206    * the radius given.
207    *
208    * @param r    the radius given
209    * @param x    the x coordinate of the point
210    * @param y    the y coordinate of the point
211    * @param z    the z coordinate of the point
212    * @param alos the list of stars to search through
213    * @return the list of stars within range inclusive.
214    */
215   public ArrayList<Star> performNaiveRadius(
216       double r, double x, double y, double z, ArrayList<Star> alos) {
217     ArrayList<Star> template = new ArrayList<>(alos);
218     template.removeIf(star -> (star.distanceTo(Arrays.asList(x, y, z)) > r));
219     template.sort(Comparator.comparingDouble(star -> star.distanceTo(Arrays.asList(x, y, z))));
220     return template;
221   }
