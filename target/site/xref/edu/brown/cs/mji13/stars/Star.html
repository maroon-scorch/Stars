1   package edu.brown.cs.mji13.stars;
2   
3   import edu.brown.cs.mji13.kdTree.HasCoordinates;
4   import edu.brown.cs.mji13.validations.StringValFunctions;
5   
6   import java.util.ArrayList;
7   import java.util.List;
8   
9   /**
10   * The Stars Object that stores the information from the CSV File.
11   */
12  public class Star implements StringValFunctions, HasCoordinates {
13    /**
14     * The list of fields Stars stores:
15     * starID, star name, ArrayList of coordinates.
16     */
17    private String starID, starName;
18    private ArrayList<Double> coordinates;
19  
20    /**
21     * Creates a Star Object with the String Parameters Given.
22     *
23     * @param starID      id of Star
24     * @param starName    name of Star
25     * @param coordinates  the arbitrary list of coordinates given
26     */
27    public Star(String starID, String starName, List<Double> coordinates) {
28      if (coordinates.size() < 1) {
29        throw new IllegalArgumentException("Stars cannot have a dimension less than 1");
30      }
31      this.starID = starID;
32      this.starName = starName;
33      this.coordinates = new ArrayList<>(coordinates);
34    }
35  
36  
37    /**
38     * Returns the dimension of the Coordinates in Stars.
39     *
40     * @return the Star's dimension (size)
41     */
42    public int numDimensions() {
43      return coordinates.size();
44    }
45  
46    /**
47     * Returns the coordinate at the index of the Coordinates.
48     *
49     * @param d - the index of the coordinate
50     * @return the value at said index if index given is not out of bounds.
51     */
52    public double getCoordinate(int d) {
53      if (d < 0) {
54        throw new IndexOutOfBoundsException("Can't Retrieve a Coordinate with Index below Zero");
55      }
56  
57      if (d >= coordinates.size()) {
58        throw new IndexOutOfBoundsException("Can't Retrieve a Coordinate with "
59            + "Index above its Dimension");
60      }
61      return coordinates.get(d);
62    }
63  
64    /**
65     * Returns the entire coordinate list.
66     *
67     * @return the arrayList of doubles.
68     */
69    public ArrayList<Double> getCoordinates() {
70      return coordinates;
71    }
72  
73    /**
74     * Gets the distance of the star to a coordinate given.
75     *
76     * @param cords - the Coordinate given
77     * @return the distance if the coordinates given and the star's coordinate have the same
78     * dimension (share same size).
79     */
80    public double distanceTo(List<Double> cords) {
81      if (cords.size() != coordinates.size()) {
82        throw new IllegalArgumentException("Cannot find distance between two coordinates"
83            + "because of inconsistent dimensions.");
84      }
85  
86      double sum = 0;
87      for (int i = 0; i < cords.size(); i++) {
88        double diff = coordinates.get(i) - cords.get(i);
89        sum += Math.pow(diff, 2);
90      }
91  
92      return Math.sqrt(sum);
93    }
94  
95    /**
96     * Gets the name of the star.
97     *
98     * @return name
99     */
100   public String getName() {
101     return starName;
102   }
103 
104   /**
105    * Gets the id of the star.
106    *
107    * @return name
108    */
109   public String getStarID() {
110     return starID;
111   }
112 
113   /**
114    * Enables the star object to be printable.
115    *
116    * @return the String Literal of the star (which is its id)
117    */
118   public String toString() {
119     return starID;
120   }
121 
122   /**
123    * Enables the star object to be printable.
124    *
125    * @return the HTML Literal of the Star, which is a row of its information.
126    */
127   public String toHTML() {
128     return "<tr>" + "<td>" + starID + "</td>" + "<td>" + starName + "</td>"
129         + "<td>" + coordinates.toString() + "</td>" + "</tr>";
130   }
131 
132 }
