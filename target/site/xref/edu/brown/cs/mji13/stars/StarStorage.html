1   package edu.brown.cs.mji13.stars;
2   
3   import edu.brown.cs.mji13.kdTree.KDTree;
4   
5   import java.util.ArrayList;
6   import java.util.HashMap;
7   import java.util.Map;
8   import java.util.Optional;
9   import java.util.Set;
10  
11  /**
12   * Stars Storage Object for storing common datatypes shared between all stars commands.
13   */
14  public class StarStorage {
15    /**
16     * The argument for StarsStorage: a name assigned to the StorageUnit.
17     */
18    private String title;
19  
20    /**
21     * Constructs a simple StarStorage Object.
22     *
23     * @param title - the name assigned to the object.
24     */
25    public StarStorage(String title) {
26      this.title = title;
27    }
28  
29    /**
30     * The name of the current file the Command is operating on.
31     */
32    private final StringBuilder currentFile = new StringBuilder();
33  
34    /**
35     * The list of Stars to store the converted lines to stars in.
36     */
37    private final ArrayList<Star> starsList = new ArrayList<>();
38  
39    /**
40     * The KDTree of Stars to store the convert the list of stars to tree.
41     */
42    private final KDTree<Star> starsTree = new KDTree<>(3);
43  
44    /**
45     * The Hashmap of the Star's Name to the Star Object.
46     */
47    private Map<String, Star> starsMap = new HashMap<>();
48  
49    /**
50     * Returns the name of the current file the stars commands are operating on.
51     *
52     * @return the current file name
53     */
54    public String getFileName() {
55      return currentFile.toString();
56    }
57  
58    /**
59     * Changes the name of the current file the stars commands are operating on.
60     *
61     * @param name - the name to replace the current name.
62     */
63    public void setName(String name) {
64      currentFile.replace(0, currentFile.length(), name);
65    }
66  
67    /**
68     * Returns the current list of stars that the stars commands are operating on.
69     *
70     * @return the starsList.
71     */
72    public ArrayList<Star> getStarsList() {
73      return new ArrayList<>(starsList);
74    }
75  
76    /**
77     * Replaces the current list of stars with another list of stars.
78     *
79     * @param slist - the list of stars to replace the current one.
80     */
81    public void setList(ArrayList<Star> slist) {
82      starsList.clear();
83      starsList.addAll(slist);
84    }
85  
86    /**
87     * Returns the current KD Tree of Stars.
88     *
89     * @return the current KD Tree of Stars.
90     */
91    public KDTree<Star> getStarsTree() {
92      return starsTree;
93    }
94  
95    /**
96     * Takes in a list of stars and uses it to construct a KD Tree of stars.
97     *
98     * @param slist - the list of stars to be used to construct the KD Tree.
99     */
100   public void setListToTree(ArrayList<Star> slist) {
101     starsTree.clear();
102     starsTree.constructTree(slist);
103   }
104 
105   /**
106    * Given a name, searches if the Hashmap contains a star with said name.
107    *
108    * @param name - the key to find the corresponding star.
109    * @return an Optional of stars if a star with the name is found, an Optional of empty if
110    * no such star is found.
111    */
112   public Optional<Star> getStarFromMap(String name) {
113     if (starsMap.containsKey(name)) {
114       return Optional.of(starsMap.get(name));
115     }
116     return Optional.empty();
117   }
118 
119   /**
120    * Returns the set of all names stored.
121    *
122    * @return the set of all names stored.
123    */
124   public Set<String> getStarNames() {
125     return starsMap.keySet();
126   }
127 
128   /**
129    * Takes in a list of stars and uses it to construct a Hashmap of the star's name
130    * to the stars. If the name is empty, do not add the star.
131    *
132    * @param slist - the list of stars to be used to construct the Hashmap.
133    */
134   public void setListToStarsMap(ArrayList<Star> slist) {
135     starsMap.clear();
136     for (Star item : slist) {
137       String name = item.getName();
138       if (!name.isEmpty()) {
139         starsMap.put(name, item);
140       }
141     }
142   }
143 
144 }
