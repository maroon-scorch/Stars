8   public interface HasCoordinates {

13     */
14    int numDimensions();
15  
16    /**
17     * Given an index d, returns the coordinate at that index.
18     *
19     * @param d - the index given
20     * @return the coordinate at said index
21     */

23  
24    /**
25     * Given another configuration of coordinates, find the distance between this
26     * coordinate and that.
27     *
28     * @param cords - the coordinates given
29     * @return the distance between the two coordinates.
30     */
31    double distanceTo(List<Double> cords);
